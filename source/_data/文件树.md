// 1. 配置存储与状态恢复
const STORAGE_KEY = "obsidian-nav-tree-state";
const savedState = JSON.parse(localStorage.getItem(STORAGE_KEY) || "{}");

const updateState = (id, isOpen) => {
    savedState[id] = isOpen;
    localStorage.setItem(STORAGE_KEY, JSON.stringify(savedState));
};

// 2. 数据获取与逻辑树构建
const pages = dv.pages('"_posts"').where(p => p.categories);
const root = {};

for (let page of pages) {
    let cats = Array.isArray(page.categories) ? page.categories : [page.categories];
    let current = root;
    for (let i = 0; i < cats.length; i++) {
        let cat = cats[i];
        if (!current[cat]) {
            current[cat] = { _files: [], _sub: {} };
        }
        if (i === cats.length - 1) {
            current[cat]._files.push(page);
        }
        current = current[cat]._sub;
    }
}

// 3. 递归渲染函数
function renderTree(tree, container, parentPath = "") {
    const catNames = Object.keys(tree);
    
    catNames.forEach(name => {
        const currentPath = parentPath ? `${parentPath}/${name}` : name;
        const details = container.createEl("details");
        
        if (savedState[currentPath]) {
            details.setAttribute("open", "true");
        }

        details.addEventListener("toggle", () => {
            updateState(currentPath, details.open);
        });

        const summary = details.createEl("summary");
        summary.setText("📂 " + name);
        summary.style.cursor = "pointer";
        summary.style.fontWeight = "bold";
        summary.style.padding = "5px 0";
        summary.style.listStyle = "none"; 

        const content = details.createEl("div");
        content.style.paddingLeft = "22px";
        content.style.borderLeft = "1px solid var(--background-modifier-border)"; 

        // --- 核心排序逻辑 ---
        const files = tree[name]._files;
        if (files && files.length > 0) {
            // 按文件创建时间 (ctime) 倒序排列：最新的排在前面
            // 如果想正序排列，把 b 和 a 的位置互换即可
            files.sort((a, b) => b.file.ctime - a.file.ctime);

            files.forEach(page => {
                const fileEl = content.createEl("div");
                fileEl.style.padding = "4px 0";
                dv.el("span", "📄 ", { container: fileEl });
                dv.el("span", page.file.link, { container: fileEl });
            });
        }

        if (Object.keys(tree[name]._sub).length > 0) {
            renderTree(tree[name]._sub, content, currentPath);
        }
    });
}

// 4. 清空并执行
dv.container.empty();
renderTree(root, dv.container);
